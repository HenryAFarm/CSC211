import javax.swing.JOptionPane;

// first class (parent)
class GradedActivity {
    private double score;

    public void setScore(double s) {
        score = s;
    }

    public double getScore() {
        return score;
    }

    public char getGrade() {
        char letterGrade;

        if (score >= 90)
            letterGrade = 'A';
        else if (score >= 80)
            letterGrade = 'B';
        else if (score >= 70)
            letterGrade = 'C';
        else if (score >= 60)
            letterGrade = 'D';
        else
            letterGrade = 'F';

        return letterGrade;
    }
}

// second class
class PassFailActivity extends GradedActivity {
    private double minPassingScore;

    public PassFailActivity(double mps) {
       minPassingScore = mps;
    }
}

// third class
class PassFailExam extends PassFailActivity {
    private int numQuestions;   // Number of questions
    private double pointsEach;  // Points for each question
    private int numMissed;      // number missed

    public PassFailExam(int questions, int missed, double minPassing) {
        // Call the superclass constructor.
        super(minPassing);

        // Declare a local variable for the score.
        double numericScore;

        // Set the numQuestions and numMissed fields.
        numQuestions = questions;
        numMissed = missed;

        // Calculate the points for each question and
        // the numeric score for this exam.
        pointsEach = 100.0 / questions;
        numericScore = 100.0 - (missed * pointsEach);

        // Call the superclass's setScore method to
        // set the numeric score.
        setScore(numericScore);
    }

    public double getPointsEach() {
        return pointsEach;
    }

    public int getNumMissed() {
        return numMissed;
    }
}

// final class
public class Main {
    public static void main(String[] args) {
        String input;           // To hold user input
        int questions;          // Number of questions
        int missed;             // Number of questions missed
        double minPassing;      // Minimum passing score
        String output;          // To hold the output message

        // 1. Get the number of questions on the exam
        input = JOptionPane.showInputDialog("How many questions are on the exam?");
        // Convert the String input to an int.
        questions = Integer.parseInt(input);

        // 2. Get the number of questions missed
        input = JOptionPane.showInputDialog("How many questions did the student miss?");
        // Convert the String input to an int.
        missed = Integer.parseInt(input);

        // 3. Get the minimum passing score
        input = JOptionPane.showInputDialog("What is the minimum passing score?");
        // Convert the String input to a double.
        minPassing = Double.parseDouble(input);

        
        PassFailExam exam = new PassFailExam(questions, missed, minPassing);
        //the output message.
        output = String.format("Each question counts %.2f points.\n", exam.getPointsEach()) +
                String.format("The exam score is %.2f\n", exam.getScore()) +
                "The exam grade is " + exam.getGrade();


        
        JOptionPane.showMessageDialog(null, output);
        System.exit(0);
    }
}
