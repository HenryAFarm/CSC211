import java.util.Scanner;
public class FibonacciExample {

  //The recursive way 
    public static long fibonacciRecursive(int n) {
        // Base cases
        if (n <= 1) {
            return n; 
        }
        
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }

    //The Iterative way of doing Fibonacci
    public static long fibonacciIterative(int n) {
        // Base cases
        if (n <= 1) {
            return n;
        }

        long current = 0; 
        long next = 1;    
        
        for (int i = 2; i <= n; i++) {
            long temp = current;
            current = next;    
            next = temp + current; 
        }
        
 
        return next;
    }

    public static void main(String[] args) {
    int n;
    Scanner scnr = new Scanner(System.in);
    
        System.out.println("-----Fibonacci Sequence-----" );
        System.out.print("Enter a integer number: ");
        n = scnr.nextInt();
        
        
        //Demonstration of Recursive method
            long resultRecursive = fibonacciRecursive(n);
            System.out.println("Recursive Result (F(" + n + ")): " + resultRecursive);

        
        // Demonstration of Iterative method
            long resultIterative = fibonacciIterative(n);
            System.out.println("Iterative Result (F(" + n + ")): " + resultIterative);
    
        }

    }
